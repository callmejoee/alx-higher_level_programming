#!/usr/bin/python3
import unittest
from 0-add_integer import add_integer

class TestAddInteger(unittest.TestCase):
    def test_normal_numbers(self):
        self.assertEqual(add_integer(1, 2), 3)
        self.assertEqual(add_integer(-1, -2), -3)
        self.assertEqual(add_integer(0, 0), 0)
        self.assertEqual(add_integer(1.0, 2.0), 3)
        self.assertEqual(add_integer(1.2, 2.5), 3)
        self.assertEqual(add_integer(-1.2, -2.5), -3)
        self.assertEqual(add_integer(5, 10), 15)
        self.assertEqual(add_integer(-5, 10), 5)
        self.assertEqual(add_integer(-5, -10), -15)

    def test_string(self):
        with self.assertRaises(TypeError) as cm:
            add_integer(4, "School")
        self.assertEqual(str(cm.exception), "a must be an integer")

        with self.assertRaises(TypeError) as cm:
            add_integer(None)
        self.assertEqual(str(cm.exception), "a must be an integer")
